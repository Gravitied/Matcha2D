In matcha_physics.c, rewrite the m2_get_contacts function body:

```c
WASM_EXPORT void m2_get_contacts(
    int worldHandle,
    int *bodyA, int *bodyB,
    float *nx, float *ny,
    float *localAX, float *localAY,
    float *localBX, float *localBY,
    float *penetration,
    int *contactsPerManifold,
    int maxManifolds)
{
    WorldState *ws = get_world(worldHandle);
    if (!ws) return;
    int n = ws->nManifolds < maxManifolds ? ws->nManifolds : maxManifolds;
    for (int i = 0; i < n; i++) {
        const Manifold *m = &ws->manifolds[i];
        bodyA[i]              = m->bodyA;
        bodyB[i]              = m->bodyB;
        nx[i]                 = m->normalX;
        ny[i]                 = m->normalY;
        contactsPerManifold[i] = m->contactCount;
        if (m->contactCount > 0) {
            localAX[i]      = m->contacts[0].localAx;
            localAY[i]      = m->contacts[0].localAy;
            localBX[i]      = m->contacts[0].localBx;
            localBY[i]      = m->contacts[0].localBy;
            penetration[i]  = m->contacts[0].penetration;
        } else {
            localAX[i] = localAY[i] = localBX[i] = localBY[i] = penetration[i] = 0.0f;
        }
    }
}
```

Note: The current broken code used px/py for a world-space attempt and only wrote contacts[0].localBx into px[i]. The new code writes all four local-space coords per manifold into their own arrays. The TypeScript side (Plan 02) will then reconstruct world-space from localA using mat2FromAngle.
