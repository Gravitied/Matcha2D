function makeBuffers(capacity = 16) { return createBuffers(capacity) }

function placeBox(buf: ReturnType<typeof makeBuffers>, idx: number,
  x: number, y: number, hx: number, hy: number, angle = 0) {
  buf.positionX[idx] = x; buf.positionY[idx] = y; buf.angle[idx] = angle
  buf.halfExtentX[idx] = hx; buf.halfExtentY[idx] = hy
  buf.shapeType[idx] = ShapeType.Box; buf.flags[idx] = BodyFlags.ACTIVE
}

function placeCircle(buf: ReturnType<typeof makeBuffers>, idx: number,
  x: number, y: number, radius: number) {
  buf.positionX[idx] = x; buf.positionY[idx] = y; buf.shapeRadius[idx] = radius
  buf.shapeType[idx] = ShapeType.Circle; buf.flags[idx] = BodyFlags.ACTIVE
}

function placePolygon(buf: ReturnType<typeof makeBuffers>, idx: number,
  x: number, y: number, vertices: Array<[number, number]>, angle = 0) {
  buf.positionX[idx] = x; buf.positionY[idx] = y; buf.angle[idx] = angle
  buf.shapeType[idx] = ShapeType.Polygon
  buf.shapeVertexCount[idx] = vertices.length
  buf.flags[idx] = BodyFlags.ACTIVE

    buf.shapeVerticesY[base + i] = vertices[i][1]
  }
}

function pair(a: number, b: number) {
  return { a: a as BodyHandle, b: b as BodyHandle }

function vecLen(x: number, y: number) { return Math.sqrt(x*x + y*y) }

}

  const len = vecLen(v.x, v.y)

}
