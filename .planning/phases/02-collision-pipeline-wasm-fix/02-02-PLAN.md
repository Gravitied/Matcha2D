4. Rewrite getContacts() body in the instance object:
```typescript
getContacts(worldHandle): ContactManifold[] {
  const contactCount = m2GetContactCount(worldHandle)
  if (contactCount === 0) return []

  const pBodyA            = allocAndCopy(Module, new Int32Array(contactCount),   contactCount)
  const pBodyB            = allocAndCopy(Module, new Int32Array(contactCount),   contactCount)
  const pNx               = allocAndCopy(Module, new Float32Array(contactCount), contactCount)
  const pNy               = allocAndCopy(Module, new Float32Array(contactCount), contactCount)
  const pLocalAX          = allocAndCopy(Module, new Float32Array(contactCount), contactCount)
  const pLocalAY          = allocAndCopy(Module, new Float32Array(contactCount), contactCount)
  const pLocalBX          = allocAndCopy(Module, new Float32Array(contactCount), contactCount)
  const pLocalBY          = allocAndCopy(Module, new Float32Array(contactCount), contactCount)
  const pPen              = allocAndCopy(Module, new Float32Array(contactCount), contactCount)
  const pContactsPerMfold = allocAndCopy(Module, new Int32Array(contactCount),   contactCount)

  m2GetContacts(worldHandle,
    pBodyA, pBodyB,
    pNx, pNy,
    pLocalAX, pLocalAY,
    pLocalBX, pLocalBY,
    pPen, pContactsPerMfold,
    contactCount)

  const heapBodyA    = new Int32Array(Module.HEAPU8.buffer,  pBodyA,            contactCount)
  const heapBodyB    = new Int32Array(Module.HEAPU8.buffer,  pBodyB,            contactCount)
  const heapNx       = new Float32Array(Module.HEAPU8.buffer, pNx,              contactCount)
  const heapNy       = new Float32Array(Module.HEAPU8.buffer, pNy,              contactCount)
  const heapLocalAX  = new Float32Array(Module.HEAPU8.buffer, pLocalAX,         contactCount)
  const heapLocalAY  = new Float32Array(Module.HEAPU8.buffer, pLocalAY,         contactCount)
  const heapLocalBX  = new Float32Array(Module.HEAPU8.buffer, pLocalBX,         contactCount)
  const heapLocalBY  = new Float32Array(Module.HEAPU8.buffer, pLocalBY,         contactCount)
  const heapPen      = new Float32Array(Module.HEAPU8.buffer, pPen,             contactCount)

  const manifolds: ContactManifold[] = []
  for (let i = 0; i < contactCount; i++) {
    manifolds.push({
      bodyA: heapBodyA[i] as BodyHandle,
      bodyB: heapBodyB[i] as BodyHandle,
      normal: { x: heapNx[i], y: heapNy[i] },
      contacts: [{
        localA: { x: heapLocalAX[i], y: heapLocalAY[i] },
        localB: { x: heapLocalBX[i], y: heapLocalBY[i] },
        penetration: heapPen[i],
      }],
    })
  }

  freeAll(Module, [pBodyA, pBodyB, pNx, pNy, pLocalAX, pLocalAY, pLocalBX, pLocalBY, pPen, pContactsPerMfold])
  return manifolds
},
```
